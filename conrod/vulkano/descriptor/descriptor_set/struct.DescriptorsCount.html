Number of available descriptors slots in a pool.

Example

use vulkano::descriptor::descriptor_set::DescriptorsCount;

let _descriptors = DescriptorsCount {
    uniform_buffer: 10,
    input_attachment: 5,
    .. DescriptorsCount::zero()
};
